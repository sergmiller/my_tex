    –í–∞–∂–Ω–æ–π —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫–æ–π –∑–≤—É–∫–æ–≤—ã—Ö –≤–æ–ª–Ω —è–≤–ª—è–µ—Ç—Å—è —Å–∫–æ—Ä–æ—Å—Ç—å –∏—Ö
    —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–∏—è. –û–Ω–∞ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è –∏–Ω–µ—Ä—Ü–∏–æ–Ω–Ω—ã–º–∏ –∏ —É–ø—Ä—É–≥–∏–º–∏ —Å–≤–æ–π—Å—Ç–≤–∞–º–∏ —Å—Ä–µ–¥—ã. –°–∫–æ—Ä–æ—Å—Ç—å —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–∏—è –ø—Ä–æ–¥–æ–ª—å–Ω—ã—Ö –≤–æ–ª–Ω –≤ –±–µ–∑–≥—Ä–∞–Ω–∏—á–Ω–æ–π –æ–¥–Ω–æ—Ä–æ–¥–Ω–æ–π —Å—Ä–µ–¥–µ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è –≤—ã—Ä–∞–∂–µ–Ω–∏–µ–º:
     
    \begin{equation}

    \end{equation}

    –î–∞–≤–ª–µ–Ω–∏–µ $P$ –∑–∞–≤–∏—Å–∏—Ç –Ω–µ —Ç–æ–ª—å–∫–æ –æ—Ç –ø–ª–æ—Ç–Ω–æ—Å—Ç–∏ $\rho$, –Ω–æ –∏ –æ—Ç —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã $T$ . –ü–æ—ç—Ç–æ–º—É –Ω—É–∂–Ω–æ —É—Ç–æ—á–Ω–∏—Ç—å, –≤ –∫–∞–∫–æ–º —Å–º—ã—Å–ª–µ –ø–æ–Ω–∏–º–∞–µ—Ç—Å—è –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—è $\frac{dP}{d\rho}$.
    –ö–æ–ª–µ–±–∞–Ω–∏—è –ø–ª–æ—Ç–Ω–æ—Å—Ç–∏ –∏ —Å–≤—è–∑–∞–Ω–Ω—ã–µ —Å –Ω–∏–º–∏ –∫–æ–ª–µ–±–∞–Ω–∏—è —Ç–µ–º–ø–µ—Ä–∞- —Ç—É—Ä—ã –≤ –∑–≤—É–∫–æ–≤–æ–π –≤–æ–ª–Ω–µ –ø—Ä–æ–∏—Å—Ö–æ–¥—è—Ç –Ω–∞—Å—Ç–æ–ª—å–∫–æ –±—ã—Å—Ç—Ä–æ, –∞ —Ç–µ–ø–ª–æ–ø—Ä–æ–≤–æ–¥–Ω–æ—Å—Ç–∏ –≥–∞–∑–æ–≤ –Ω–∞—Å—Ç–æ–ª—å–∫–æ –º–∞–ª—ã, —á—Ç–æ –¥–ª—è —Ç–∞–∫–∏—Ö –ø—Ä–æ—Ü–µ—Å—Å–æ–≤ —Ç–µ–ø–ª–æ–æ–±–º–µ–Ω–æ–º –º–æ–∂–Ω–æ –ø—Ä–µ–Ω–µ–±—Ä–µ—á—å, —Ç–∞–∫ —á—Ç–æ –ø—Ä–æ—Ü–µ—Å—Å —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–∏—è –∑–≤—É–∫–∞ –º–æ–∂–Ω–æ —Å—á–∏—Ç–∞—Ç—å \textit{–∞–¥–∏–∞–±–∞—Ç–∏—á–µ—Å–∫–∏–º}. –°–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ, –ø—Ä–æ–∏–∑–≤–æ–¥–Ω—É—é $\frac{dP}{d\rho}$ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ —Ä–∞—Å—Å—á–∏—Ç—ã–≤–∞—Ç—å –¥–ª—è –∞–¥–∏–∞–±–∞—Ç–∏—á–µ—Å–∫–æ–≥–æ –ø—Ä–æ—Ü–µ—Å—Å–∞.


    \textbf{ –ü–µ—Ä–≤–æ–µ –Ω–∞—á–∞–ª–æ —Ç–µ—Ä–º–æ–¥–∏–Ω–∞–º–∏–∫–∏.}

    –ò–∑ –∑–∞–∫–æ–Ω–∞ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è —ç–Ω–µ—Ä–≥–∏–∏ —Å–ª–µ–¥—É–µ—Ç, —á—Ç–æ —Ç–µ–ø–ª–æ $Q$, –ø–æ–ª—É—á–µ–Ω–Ω–æ–µ —Ç–µ—Ä–º–æ–¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–π —Å–∏—Å—Ç–µ–º–æ–π, —Ä–∞—Å—Ö–æ–¥—É–µ—Ç—Å—è –Ω–∞ –∏–∑–º–µ–Ω–µ–Ω–∏–µ –µ—ë –≤–Ω—É—Ç—Ä–µ–Ω–Ω–µ–π —ç–Ω–µ—Ä–≥–∏–∏ $\Delta U$ –∏ –Ω–∞ —Å–æ–≤–µ—Ä—à–µ–Ω–∏–µ —Ä–∞–±–æ—Ç—ã $A$ –Ω–∞–¥ –≤–Ω–µ—à–Ω–∏–º–∏ —Ç–µ–ª–∞–º–∏:
    \begin{equation}

    \end{equation}
    –î–ª—è –±–µ—Å–∫–æ–Ω–µ—á–Ω–æ –º–∞–ª–æ–≥–æ –ø—Ä–æ—Ü–µ—Å—Å–∞ —É—Ä–∞–≤–Ω–µ–Ω–∏–µ (4) –ø—Ä–∏–Ω–∏–º–∞–µ—Ç –≤–∏–¥

    \end{equation}

    –ü–æ—Å–∫–æ–ª—å–∫—É –≤–Ω—É—Ç—Ä–µ–Ω–Ω—è—è —ç–Ω–µ—Ä–≥–∏—è —è–≤–ª—è–µ—Ç—Å—è —Ñ—É–Ω–∫—Ü–∏–µ–π —Å–æ—Å—Ç–æ—è–Ω–∏—è —Å–∏—Å—Ç–µ–º—ã, –¥–ª—è –µ—ë —ç–ª–µ–º–µ–Ω—Ç–∞—Ä–Ω–æ–≥–æ –ø—Ä–∏—Ä–∞—â–µ–Ω–∏—è –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω –∑–Ω–∞–∫ –ø–æ–ª–Ω–æ–≥–æ –¥–∏—Ñ—Ñ–µ—Ä–µ–Ω—Ü–∏–∞–ª–∞ $dU$ , –∞ –ø—Ä–∏—Ä–∞—â–µ–Ω–∏—è –∏ —Ç–µ–ø–ª–∞, –∏ —Ä–∞–±–æ—Ç—ã –Ω–µ —è–≤–ª—è—é—Ç—Å—è –ø–æ–ª–Ω—ã–º–∏ –¥–∏—Ñ—Ñ–µ—Ä–µ–Ω—Ü–∏–∞–ª–∞–º–∏, –∞ $Q$ –∏ $A$ ‚Äî –Ω–µ —Ñ—É–Ω–∫—Ü–∏–∏ —Å–æ—Å—Ç–æ—è–Ω–∏—è(
    –î–ª—è –∏–Ω—Ç–µ–≥—Ä–∏—Ä–æ–≤–∞–Ω–∏—è –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –∑–∞–¥–∞–Ω –≤–µ—Å—å –ø—Ä–æ–º–µ–∂—É—Ç–æ—á–Ω—ã–π –ø—Ä–æ—Ü–µ—Å—Å, –ø–æ—Å–∫–æ–ª—å–∫—É —Ä–µ–∑—É–ª—å—Ç–∞—Ç –±—É–¥–µ—Ç –∑–∞–≤–∏—Å–µ—Ç—å –æ—Ç –µ–≥–æ –≤–∏–¥–∞, –∞ –Ω–µ —Ç–æ–ª—å–∫–æ –æ—Ç –Ω–∞—á–∞–ª—å–Ω–æ–≥–æ –∏ –∫–æ–Ω–µ—á–Ω–æ–≥–æ —Å–æ—Å—Ç–æ—è–Ω–∏–π).

    \textbf{ –†–∞–±–æ—Ç–∞ –≥–∞–∑–∞.}

    –†–∞—Å—Å–º–æ—Ç—Ä–∏–º —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–µ –≥–∞–∑–∞ –≤ —Ü–∏–ª–∏–Ω–¥—Ä–µ, –∑–∞–∫—Ä—ã—Ç–æ–º –ø–æ–¥–≤–∏–∂–Ω—ã–º –ø–æ—Ä—à–Ω–µ–º. –ù–∞ –ø–æ—Ä—à–µ–Ω—å –¥–µ–π—Å—Ç–≤—É–µ—Ç —Å–∏–ª–∞ $F$ , —Ä–∞–≤–Ω–∞—è –ø—Ä–æ–∏–∑–≤–µ–¥–µ–Ω–∏—é –¥–∞–≤–ª–µ–Ω–∏—è –≥–∞–∑–∞ $F$ –Ω–∞ –ø–ª–æ—â–∞–¥—å –ø–æ—Ä—à–Ω—è $S$. –ü—Ä–∏ —Å–º–µ—â–µ–Ω–∏–∏ –Ω–∞ –º–∞–ª—É—é –≤–µ–ª–∏—á–∏–Ω—É $dx$ –≥–∞–∑ —Å–æ–≤–µ—Ä—à–∞–µ—Ç —Ä–∞–±–æ—Ç—É
    \begin{equation}
        \delta A = F dx = P S dx = P dV.
    \end{equation}
    –≥–¥–µ $dV$  ‚Äî –º–∞–ª–æ–µ –∏–∑–º–µ–Ω–µ–Ω–∏–µ –æ–±—ä—ë–º–∞ –≥–∞–∑–∞.
    –ó–Ω–∞—á–∏—Ç –ø–æ–ª–Ω–∞—è —Ä–∞–±–æ—Ç–∞ –ø—Ä–∏ –Ω–µ–∫–æ—Ç–æ—Ä–æ–º –ø—Ä–æ—Ü–µ—Å—Å–µ –∏–º–µ–µ—Ç –≤–∏–¥:
    \begin{equation}
        A = \int P(V) dV
    \end{equation}
    –ê –ø–µ—Ä–≤–æ–µ –Ω–∞—á–∞–ª–æ —Ç–µ—Ä–º–æ–¥–∏–Ω–∞–º–∏–∫–∏ –¥–ª—è –≥–∞–∑–æ–≤ –ø–æ—Å–ª–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è —Ñ–æ—Ä–º—É–ª—ã (6) –±—É–¥–µ—Ç –∏–º–µ—Ç—å –≤–∏–¥:
    \begin{equation}
        \delta Q = dU + P dV
    \end{equation}

    \textbf{–¢–µ–ø–ª–æ–µ–º–∫–æ—Å—Ç—å}
    –û—Ç–Ω–æ—à–µ–Ω–∏–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ —Ç–µ–ø–ª–∞ $\delta Q$, –ø–æ–≥–ª–æ—â—ë–Ω–Ω–æ–≥–æ $\nu$ –º–æ–ª—è–º–∏ –≥–∞–∑–∞ –ø—Ä–∏ –Ω–µ–∫–æ—Ç–æ—Ä–æ–º –ø—Ä–æ—Ü–µ—Å—Å–µ, –∫–æ—Ç–æ—Ä—ã–π –æ–±–æ–∑–Ω–∞—á–∏–º –∏–Ω–¥–µ–∫—Å–æ–º $x$, –∫ –ø–æ–≤—ã—à–µ–Ω–∏—é –µ–≥–æ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã –Ω–∞ $dT$, –¥–µ–ª—ë–Ω–Ω–æ–µ –Ω–∞ —á–∏—Å–ª–æ –º–æ–ª–µ–π $\nu$, –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è \textit{–º–æ–ª—è—Ä–Ω–æ–π —Ç–µ–ø–ª–æ–µ–º–∫–æ—Å—Ç—å—é –≥–∞–∑–∞}:

    \begin{equation}
        C = \bigg(\frac{\delta Q}{dT}\bigg)_x / \nu
    \end{equation}

    –ë—É–¥–µ–º —Å—á–∏—Ç–∞—Ç—å —á—Ç–æ $U = U(V,T)$ (—Ç–∞–∫ –∫–∞–∫ —Å–∏—Å—Ç–µ–º–∞ –æ–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è —Ç—Ä–µ–º—è –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º–∏: $V,T,P$ –Ω–æ –ø—Ä–∏ —ç—Ç–æ–º –µ—Å—Ç—å —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –ú–µ–Ω–¥–µ–ª–µ–µ–≤–∞-–ö–ª–∞–ø–µ–π—Ä–æ–Ω–∞: $P = P(V,T)$).
    –ó–Ω–∞—á–∏—Ç –ø–æ–ª–Ω—ã–π –¥–∏—Ñ—Ñ–µ—Ä–µ–Ω—Ü–∏–∞–ª –¥–ª—è $U$ –∏–º–µ–µ—Ç —Å–ª–µ–¥—É—é—â–∏–π –≤–∏–¥:

    \begin{equation}
        dU = \bigg(\frac{\delta U}{\delta T}\bigg)_V dT + \bigg(\frac{\delta U}{\delta V}\bigg)_T dV
    \end{equation}

    –ü–æ–¥—Å—Ç–∞–≤–∏–º –µ–≥–æ –≤ –ø–µ—Ä–≤–æ–µ –Ω–∞—á–∞–ª–æ —Ç–µ—Ä–º–æ–¥–∏–Ω–∞–º–∏–∫–∏:

    $  \delta Q = dU + \delta A =  \bigg(\frac{\delta U}{\delta T}\bigg)_V dT + \bigg(\frac{\delta U}{\delta V}\bigg)_T dV + PdV = \bigg(\frac{\delta U}{\delta T}\bigg)_V dT + \bigg[P + \bigg(\frac{\delta U}{\delta V}\bigg)_T \bigg] dV $

    –†–∞–∑–¥–µ–ª–∏–≤ –≤—Å–µ –Ω–∞ $dT$ –Ω–∞–π–¥–µ–º —Ç–µ–ø–ª–æ–µ–º–∫–æ—Å—Ç—å $C_x$ –≤ –ø—Ä–æ—Ü–µ—Å—Å–µ x:

    \begin{equation}
        C_x = C_v + \bigg[P + \bigg(\frac{\delta U}{\delta V}\bigg)_T \bigg] \bigg(\frac{\delta V}{\delta T}\bigg)_x
    \end{equation}

    –ó–¥–µ—Å—å –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—è $\bigg(\frac{\delta V}{\delta T}\bigg)_x$ –≤—ã—á–∏—Å–ª—è–µ—Ç—Å—è —Å —É—á—ë—Ç–æ–º –ø—Ä–æ—Ü–µ—Å—Å–∞ $x$, –ø—Ä–∏ –∫–æ—Ç–æ—Ä–æ–º –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç –ø–æ–¥–≤–æ–¥ —Ç–µ–ø–ª–∞, –Ω–∞–ø—Ä–∏–º–µ—Ä, –ø—Ä–∏ –ø–æ—Å—Ç–æ—è–Ω–Ω–æ–º –æ–±—ä—ë–º–µ $(ùë• = ùëâ )$, –ø—Ä–∏ –ø–æ—Å—Ç–æ—è–Ω–Ω–æ–º –¥–∞–≤–ª–µ–Ω–∏–∏ $(ùë• = ùëÉ )$ –∏–ª–∏ –¥—Ä—É–≥–æ–º —É—Å–ª–æ–≤–∏–∏. –í–µ–ª–∏—á–∏–Ω–∞ $C_v = \bigg(\frac{\delta U}{\delta T}\bigg)_V$ –≤ —Ñ–æ—Ä–º—É–ª–µ (11) —è–≤–ª—è–µ—Ç—Å—è —Ç–µ–ø–ª–æ—ë–º–∫–æ—Å—Ç—å—é –ø—Ä–∏ –ø–æ—Å—Ç–æ—è–Ω–Ω–æ–º –æ–±—ä—ë–º–µ(–î–µ–π—Å–≤–∏—Ç–µ–ª—å–Ω–æ, —ç—Ç–æ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ —Å–æ–≤–ø–∞–¥–∞–µ—Ç —Å –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ–º —Ç–µ–ø–ª–æ–µ–º–∫–æ—Å—Ç–∏, —Ç–∞–∫ –∫–∞–∫ –≤ –ø—Ä–æ—Ü–µ—Å—Å–µ $V = const$ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è $\delta A = 0$ –∞ –∑–Ω–∞—á–∏—Ç $\detta U = dQ$).

    –í —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º–æ–º –ø—Ä–∏–±–ª–∏–∂–µ–Ω–∏–∏ –¥–ª—è –ø–æ–∫–∞–∑–∞—Ç–µ–ª—è –∞–¥–∏–∞–±–∞—Ç—ã –≤ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–∏ —Å (13) –∏ (14) –ø–æ–ª—É—á–∏–º
